\colorbox{headerbg}{\color{white}\textbf{\ Drill Types\ }}
\vspace{1mm}
\begin{itemize}
\item \textbf{Unison} --- both hands play the same pattern.
\item \textbf{Mirror} --- one hand forward, other reverse.
\item \textbf{Contrary} --- different patterns per hand; alternate anchor.
\item \textbf{Cross-size} --- 4-fng one hand, 3-fng or 2-fng other.
\item \textbf{Independent rhythm} --- slow anchor + fast cycle; swap.
\end{itemize}

\vspace{1.5mm}
\colorbox{headerbg}{\color{white}\textbf{\ Hand Technique\ }}
\vspace{1mm}
\begin{itemize}
\item Play with \textbf{pads of fingertips}; fingers curved.
\item \textbf{Thumb} stays above the fingers.
\item Wrist \textbf{level} with forearm.
\item \textbf{Close fingers into palm} after plucking.
\item Relax between strokes.
\end{itemize}

\vspace{1.5mm}
\colorbox{headerbg}{\color{white}\textbf{\ Harp Techniques\ }}
\vspace{1mm}
\begin{itemize}
\item \textbf{Damping (étouffé)}: closed fingers mute strings between chords.
\item \textbf{Rolled chord}: fingers 4\(\to\)1 fast within one beat.
\item \textbf{Glissando}: set levers to mode, run finger across.
\item \textbf{Bisbigliando}: soft hand-tremolo on same strings.
\item \textbf{Harmonic}: palm at midpoint, thumb plucks; 8ve up, half vol.
\end{itemize}

\colorbox{headerbg}{\color{white}\textbf{\ Practice Progression\ }}
\vspace{1mm}
\begin{itemize}
\item Start \textbf{slow}: metronome \(\approx\) 40--60 bpm.
\item \textbf{4 reps} per pattern before moving on.
\item Increase by \textbf{4--8 bpm} only when clean.
\item Hands \textbf{separately first}, then combine.
\item \textbf{5--10 min daily} beats long rare sessions.
\end{itemize}
